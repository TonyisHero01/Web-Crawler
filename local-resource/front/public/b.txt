---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: mysql
  namespace: training
  labels:
    app: mysql
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mysql
  template:
    metadata:
      labels:
        app: mysql
    spec:
      containers:
        - name: mysql
		  env: 
		    - name: MYSQL_ROOT_PASSWORD
		      value: Training123@
		    - name: MYSQL_DATABASE
		      value: wpdb
		    - name: MYSQL_USER
		      value: wpuser
		    - name: MYSQL_PASSWORD
		      value: Training123@ 
          image: mysql:5.7
          ports:
            - containerPort: 3306
---
apiVersion: v1
kind: Service
metadata:
 labels:
   app: mysql
 name: mysql-service
 namespace: training
spec:
 ports:
  - port: 3306
    protocol: TCP
    targetPort: 3306
 selector:
   app: mysql
 type: ClusterIP
